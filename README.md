//Para criar a tabela utilizado no exercicio: 

create table techpoint_cliente (
    id_cliente number generated by default as identity primary key,
    nome varchar(150) not null,
    email varchar(50) not null unique,
    telefone varchar(12) 
    );

    
create table techpoint_produto (
    id_produto number generated by default as identity primary key,
    nome varchar(30) not null,
    preco number(10,2) not null,
    estoque number not null
    );
    
create table techpoint_venda (
    id_venda number generated by default as identity primary key,
    cliente_id number not null,
    produto_id number not null,
    quantidade number not null,
    preco_unitario number(10,2) not null,
    total number(10,2) not null,
    data_venda DATE not null,
    constraint FK_VENDA_CLIENTE FOREIGN key (cliente_id)
        references techpoint_cliente(id_cliente),
    constraint FK_VENDA_PRODUTO FOREIGN KEY (produto_id)
        references techpoint_produto(id_produto)
    );
